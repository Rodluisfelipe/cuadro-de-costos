rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Reglas para la colección de cotizaciones
    match /cotizaciones/{cotizacionId} {
      // LECTURA: Permitir si:
      // - Es de TECNOPHONE (siempre)
      // - O el usuario está autenticado y es el creador
      allow read: if resource.data.companyId == 'TECNOPHONE' ||
        (request.auth != null && 
         (resource.data.createdBy == request.auth.uid || 
          resource.data.userEmail == request.auth.token.email));
      
      // CREACIÓN: Permitir si:
      // - Es de TECNOPHONE Y
      // - (Usuario autenticado con datos correctos O es anónimo para aprobaciones)
      allow create: if request.resource.data.companyId == 'TECNOPHONE' &&
        (
          // Usuario autenticado normal
          (request.auth != null && 
           request.resource.data.createdBy == request.auth.uid &&
           request.resource.data.userEmail == request.auth.token.email) ||
          
          // Usuario anónimo para aprobaciones (sin auth)
          (request.auth == null && 
           request.resource.data.createdBy == 'anonymous')
        );
      
      // ACTUALIZACIÓN: Permitir si:
      // - Es de TECNOPHONE Y
      // - (Es el creador autenticado O es actualización de aprobación)
      allow update: if resource.data.companyId == 'TECNOPHONE' &&
        (
          // Usuario autenticado que es el creador
          (request.auth != null && 
           (resource.data.createdBy == request.auth.uid ||
            resource.data.userEmail == request.auth.token.email)) ||
          
          // Actualización de aprobación (cambios de status)
          (request.auth == null && 
           resource.data.companyId == 'TECNOPHONE' &&
           (request.resource.data.status in ['approved', 'revision_requested', 'denied']))
        );
      
      // ELIMINACIÓN: Solo creadores autenticados
      allow delete: if request.auth != null && 
        resource.data.createdBy == request.auth.uid;
    }
    
    // Reglas para la colección de usuarios
    match /users/{userId} {
      // LECTURA: Permitir si:
      // - Es administrador de TECNOPHONE
      // - O es el propio usuario
      allow read: if request.auth != null && 
        (
          // Administrador puede ver todos los usuarios de TECNOPHONE
          (request.auth.token.email != null && 
           exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
           get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin' &&
           get(/databases/$(database)/documents/users/$(request.auth.uid)).data.companyId == 'TECNOPHONE' &&
           resource.data.companyId == 'TECNOPHONE') ||
          
          // Usuario puede ver su propio perfil
          resource.data.firebaseUid == request.auth.uid
        );
      
      // CREACIÓN: Solo administradores de TECNOPHONE
      allow create: if request.auth != null && 
        request.resource.data.companyId == 'TECNOPHONE' &&
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin' &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.companyId == 'TECNOPHONE';
      
      // ACTUALIZACIÓN: Solo administradores o el propio usuario
      allow update: if request.auth != null && 
        resource.data.companyId == 'TECNOPHONE' &&
        (
          // Administrador puede actualizar cualquier usuario de TECNOPHONE
          (exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
           get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin' &&
           get(/databases/$(database)/documents/users/$(request.auth.uid)).data.companyId == 'TECNOPHONE') ||
          
          // Usuario puede actualizar su propio perfil (campos limitados)
          (resource.data.firebaseUid == request.auth.uid &&
           !('role' in request.resource.data.diff(resource.data).affectedKeys()) &&
           !('companyId' in request.resource.data.diff(resource.data).affectedKeys()))
        );
      
      // ELIMINACIÓN: Solo administradores
      allow delete: if request.auth != null && 
        exists(/databases/$(database)/documents/users/$(request.auth.uid)) &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == 'admin' &&
        get(/databases/$(database)/documents/users/$(request.auth.uid)).data.companyId == 'TECNOPHONE' &&
        resource.data.companyId == 'TECNOPHONE';
    }
    
    // Reglas para la colección de proveedores
    match /providers/{providerId} {
      // LECTURA: Permitir si es de TECNOPHONE
      allow read: if resource.data.companyId == 'TECNOPHONE';
      
      // CREACIÓN: Solo usuarios autenticados de TECNOPHONE
      allow create: if request.auth != null && 
        request.resource.data.createdBy == request.auth.uid &&
        request.resource.data.userEmail == request.auth.token.email &&
        request.resource.data.companyId == 'TECNOPHONE';
      
      // ACTUALIZACIÓN: Solo creadores autenticados
      allow update: if request.auth != null && 
        (resource.data.createdBy == request.auth.uid ||
         resource.data.userEmail == request.auth.token.email);
      
      // ELIMINACIÓN: Solo creadores autenticados
      allow delete: if request.auth != null && 
        resource.data.createdBy == request.auth.uid;
    }
    
    // Reglas para colecciones de prueba y conectividad
    match /connectivity-tests/{testId} {
      // Permitir operaciones de prueba
      allow read, write: if true;
    }
    
    // Reglas para otras colecciones futuras
    match /{document=**} {
      // Por defecto, solo usuarios autenticados de TECNOPHONE
      allow read, write: if request.auth != null;
    }
  }
}
